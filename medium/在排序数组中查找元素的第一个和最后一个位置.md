```go
// https://leetcode-cn.com/problems/find-first-and-last-position-of-element-in-sorted-array/
func searchRange(nums []int, target int) []int {
    if len(nums) == 0 {
        return []int{-1, -1}
    }
    if len(nums) == 1 {
        if nums[0] == target{
            return []int{0, 0}
        }
        return []int{-1, -1}
    }
    var searchLeft = func(nums []int, target int) int {
        var left, right = 0, len(nums)-1
        for {
            if left >= right {
                break
            }
            var mid = (left + right) / 2
            if nums[mid] == target { // 找到一个
                right = mid // 不会卡住
                continue
            }
            if nums[mid] < target { // 在右边
                left = mid + 1
                continue
            }
            if nums[mid] > target { // 在左边
                right = mid - 1
                continue
            }
        }
        // 无论如何都只可能是左边那个
        if left < len(nums) && nums[left] == target {
            return left
        }
        return -1
    }
    var searchRight = func(nums []int, target int) int {
        var left, right = 0, len(nums)-1
        for {
            if left >= right - 1 { // 只剩下两个
                break
            }
            var mid = (left + right) / 2
            if nums[mid] == target { // 找到一个
                left = mid // 为什么加1，因为它肯定不是最左那个
                continue
            }
            if nums[mid] < target { // 在右边
                left = mid + 1
                continue
            }
            if nums[mid] > target { // 在左边
                right = mid - 1
                continue
            }
        }
        // 可能同时出现[2,2]或者[2,1]和[1,2]
        if right >= 0 && nums[right] == target {
            return right
        }
        if left < len(nums) && nums[left] == target {
            return left
        }
        return -1
    }

    return []int{
        searchLeft(nums, target),
        searchRight(nums, target),
    }
}
```
