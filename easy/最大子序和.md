func maxSubArray(nums []int) int {
    // dp[i]代表以i结尾的最大子序和
    // dp[0] = nums[0]
    // dp[i] = max(dp[i-1]+nums[i], nums[i])
    // 注意不能一个都不选，至少选一个，因此dp[i]>=nums[i]
    var length = len(nums)
    if length == 0 {
        return 0
    }
    if length == 1 {
        return nums[0]
    }
    var res int
    var dp = make([]int, length, length)
    dp[0] = nums[0]
    res = dp[0]
    for i := 1; i < length; i++ {
        dp[i] = max(dp[i-1]+nums[i], nums[i])
        res = max(res, dp[i])
    }
    return res
}

func max(a,b int) int{
    if a >= b {
        return a
    }
    return b 
}
